# Построение эффективных агентов — практическое руководство

> **В этой серии:** [Агенты на основе LLM](Агенты%20на%20основе%20LLM.md) — [ReAct](ReAct%20-%20Reasoning%20and%20Acting.md) — [Архитектура мультиагентных систем](Архитектура%20мультиагентных%20систем.md) — [RAG](RAG.md)

## Введение

Теоретически, создание агента на основе LLM просто: дайте модели несколько инструментов, позвольте ей решить, когда их использовать, и наблюдайте, как это работает. Но в реальности всё сложнее. В production вы столкнётесь с ограничениями по частоте запросов, стоимостью токенов, галлюцинациями модели, падениями инструментов и многим другим.

Это руководство предоставляет практические паттерны на основе исследований Anthropic и опыта индустрии по созданию агентов, которые действительно работают в production.

## Часть 1: Основные паттерны

### Паттерн 1: Простой Tool-Use Агент

Самый базовый вариант — модель выбирает, какой инструмент использовать.

```python
from anthropic import Anthropic
import json

client = Anthropic()

def basic_agent(user_query: str) -> str:
    """
    Простейший агент с инструментами.
    Модель получает список инструментов и сама выбирает, что использовать.
    """
    
    # 1. Определяем инструменты
    tools = [
        {
            "name": "get_weather",
            "description": "Get current weather for a location",
            "input_schema": {
                "type": "object",
                "properties": {
                    "location": {
                        "type": "string",
                        "description": "City name"
                    }
                },
                "required": ["location"]
            }
        },
        {
            "name": "calculate",
            "description": "Perform mathematical calculations",
            "input_schema": {
                "type": "object",
                "properties": {
                    "expression": {
                        "type": "string",
                        "description": "Mathematical expression (e.g., '2 + 2 * 3')"
                    }
                },
                "required": ["expression"]
            }
        }
    ]
    
    # 2. История сообщений
    messages = [
        {"role": "user", "content": user_query}
    ]
    
    # 3. Цикл агента
    while True:
        # Запрашиваем ответ от модели с доступом к инструментам
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            max_tokens=1024,
            tools=tools,
            messages=messages
        )
        
        # Проверяем, завершена ли работа
        if response.stop_reason == "end_turn":
            # Модель дала финальный ответ без инструментов
            for block in response.content:
                if hasattr(block, "text"):
                    return block.text
        
        # Обработка использованных инструментов
        if response.stop_reason == "tool_use":
            # Добавляем ответ ассистента в историю
            messages.append({"role": "assistant", "content": response.content})
            
            # Для каждого tool_use блока
            tool_results = []
            for block in response.content:
                if block.type == "tool_use":
                    tool_name = block.name
                    tool_input = block.input
                    
                    # Выполняем инструмент
                    if tool_name == "get_weather":
                        result = get_weather(tool_input["location"])
                    elif tool_name == "calculate":
                        result = calculate(tool_input["expression"])
                    else:
                        result = "Unknown tool"
                    
                    # Добавляем результат
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": str(result)
                    })
            
            # Добавляем результаты в историю
            messages.append({"role": "user", "content": tool_results})

def get_weather(location: str) -> str:
    """Mock weather function"""
    return f"Weather in {location}: 22°C, sunny"

def calculate(expression: str) -> float:
    """Safe calculation (в продакшене используйте eval с осторожностью)"""
    try:
        return eval(expression)
    except Exception as e:
        return f"Error: {str(e)}"

# Использование
if __name__ == "__main__":
    result = basic_agent("What's the weather in Moscow and what's 25 * 4?")
    print(result)
```

## Часть 2: Production-Grade Patterns

### Паттерн 4: Надёжный обработчик ошибок

В продакшене инструменты падают, API отвечают ошибками, сети прерываются.

```python
import asyncio
from typing import Optional, Any
from enum import Enum

class ToolError(Exception):
    """Базовый класс для ошибок инструментов"""
    pass

class RateLimitError(ToolError):
    """API превышил rate limit"""
    pass

class ToolNotFoundError(ToolError):
    """Инструмент не найден"""
    pass

class RobustToolExecutor:
    """
    Выполнитель инструментов с обработкой ошибок,
    retry логикой и логированием.
    """
    
    def __init__(self, max_retries: int = 3, backoff_factor: float = 2.0):
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
        self.tools = {}
    
    def register_tool(self, name: str, func, description: str):
        """Регистрация инструмента"""
        self.tools[name] = {
            "func": func,
            "description": description,
            "call_count": 0,
            "error_count": 0
        }
    
    async def execute(self, tool_name: str, tool_input: dict) -> Any:
        """
        Выполнить инструмент с retry логикой.
        
        Обрабатывает:
        - Rate limiting (экспоненциальный backoff)
        - Временные ошибки (повторные попытки)
        - Постоянные ошибки (graceful failure)
        """
        
        if tool_name not in self.tools:
            raise ToolNotFoundError(f"Tool '{tool_name}' not found")
        
        tool_info = self.tools[tool_name]
        tool_info["call_count"] += 1
        
        last_error = None
        
        for attempt in range(self.max_retries):
            try:
                # Выполняем инструмент
                result = await self._call_tool(tool_info["func"], tool_input)
                
                # Успех!
                return {
                    "status": "success",
                    "result": result,
                    "attempts": attempt + 1
                }
            
            except RateLimitError as e:
                last_error = e
                
                # Экспоненциальный backoff для rate limiting
                if attempt < self.max_retries - 1:
                    wait_time = (self.backoff_factor ** attempt)
                    print(f"Rate limit hit. Waiting {wait_time}s before retry...")
                    await asyncio.sleep(wait_time)
                else:
                    tool_info["error_count"] += 1
                    return {
                        "status": "failed",
                        "error": f"Rate limit after {self.max_retries} attempts",
                        "original_error": str(e)
                    }
            
            except ToolError as e:
                last_error = e
                
                # Временные ошибки — повторяем
                if attempt < self.max_retries - 1:
                    print(f"Tool error: {e}. Retrying...")
                    await asyncio.sleep(self.backoff_factor ** attempt)
                else:
                    tool_info["error_count"] += 1
                    return {
                        "status": "failed",
                        "error": str(e),
                        "attempts": attempt + 1
                    }
        
        # Финальная ошибка
        tool_info["error_count"] += 1
        return {
            "status": "failed",
            "error": f"Tool '{tool_name}' failed after {self.max_retries} attempts",
            "last_error": str(last_error)
        }
    
    async def _call_tool(self, func, tool_input):
        """Обёртка для вызова инструмента"""
        # Если асинхронный — await, иначе обычный вызов
        if asyncio.iscoroutinefunction(func):
            return await func(**tool_input)
        else:
            return func(**tool_input)
    
    def get_stats(self):
        """Получить статистику использования инструментов"""
        return {
            name: {
                "calls": info["call_count"],
                "errors": info["error_count"],
                "success_rate": (
                    (info["call_count"] - info["error_count"]) / 
                    max(1, info["call_count"])
                ) * 100
            }
            for name, info in self.tools.items()
        }
```

### Паттерн 5: Контроль стоимости токенов

Tool-use агенты дорогие. Каждый цикл = дополнительные API вызовы.

```python
class CostAwareAgent:
    """
    Агент, который отслеживает и оптимизирует стоимость токенов.
    """
    
    # Приблизительная стоимость (в долларах за 1M токенов)
    PRICING = {
        "claude-3-5-sonnet-20241022": {
            "input": 3.00,
            "output": 15.00
        },
        "claude-3-opus-20240229": {
            "input": 15.00,
            "output": 75.00
        }
    }
    
    def __init__(self, model: str = "claude-3-5-sonnet-20241022"):
        self.model = model
        self.total_input_tokens = 0
        self.total_output_tokens = 0
        self.tool_calls = 0
    
    def estimate_cost(self) -> float:
        """Оценить текущую стоимость"""
        pricing = self.PRICING.get(self.model, {
            "input": 3.00,
            "output": 15.00
        })
        
        input_cost = (self.total_input_tokens / 1_000_000) * pricing["input"]
        output_cost = (self.total_output_tokens / 1_000_000) * pricing["output"]
        
        return input_cost + output_cost
    
    def should_use_tool(self, tool_name: str, query_complexity: int) -> bool:
        """
        Решить, стоит ли использовать инструмент.
        
        Параметры:
        - tool_name: какой инструмент
        - query_complexity: сложность запроса (1-5)
        
        Возвращает True, если использование инструмента оправдано экономически.
        """
        
        # Используем простой эвристику:
        # Для простых запросов (1-2) — не используем инструменты
        # Для средних (3) — используем дешёвые инструменты
        # Для сложных (4-5) — используем всё, что нужно
        
        estimated_extra_cost = 0.0001  # $0.0001 за один tool call (приблизительно)
        current_cost = self.estimate_cost()
        
        if query_complexity <= 2:
            return False
        elif query_complexity <= 3:
            # Используем только если дешево
            return estimated_extra_cost < 0.00005
        else:
            return True
    
    def log_request(self, input_tokens: int, output_tokens: int, 
                   tool_used: bool = False):
        """Логировать использование токенов"""
        self.total_input_tokens += input_tokens
        self.total_output_tokens += output_tokens
        
        if tool_used:
            self.tool_calls += 1
        
        current_cost = self.estimate_cost()
        
        print(f"Tokens: {self.total_input_tokens + self.total_output_tokens} | "
              f"Cost: ${current_cost:.4f} | "
              f"Tool calls: {self.tool_calls}")
```

### Паттерн 6: Валидация и структурированный вывод

Модели могут галлюцинировать инструменты и параметры.

```python
from typing import Any, Dict
from pydantic import BaseModel, ValidationError

class CalculatorInput(BaseModel):
    """Валидация входа для calculator инструмента"""
    expression: str
    
    class Config:
        description = "Mathematical expression"

class SearchInput(BaseModel):
    """Валидация входа для search инструмента"""
    query: str
    max_results: int = 5
    
    class Config:
        description = "Search query"

class SafeToolValidator:
    """
    Валидирует входные данные перед выполнением инструмента.
    Предотвращает injection атаки и hallucinations.
    """
    
    def __init__(self):
        self.validators = {
            "calculator": CalculatorInput,
            "search": SearchInput,
        }
    
    def validate_tool_call(self, tool_name: str, tool_input: Dict) -> bool:
        """
        Проверить, валидны ли параметры инструмента.
        
        Возвращает True если OK, False если некорректно.
        """
        
        if tool_name not in self.validators:
            return False
        
        try:
            validator = self.validators[tool_name]
            validator(**tool_input)
            return True
        except ValidationError as e:
            print(f"Validation error for '{tool_name}': {e}")
            return False
    
    def sanitize_input(self, tool_name: str, tool_input: Dict) -> Dict:
        """
        Очистить входные данные перед использованием.
        
        Заменяет потенциально опасные значения на безопасные.
        """
        
        if tool_name == "calculator":
            # Разрешаем только безопасные символы
            expr = tool_input.get("expression", "")
            safe_expr = "".join(
                c for c in expr 
                if c in "0123456789+-*/(). "
            )
            return {"expression": safe_expr}
        
        elif tool_name == "search":
            # Ограничиваем max_results
            max_results = min(
                tool_input.get("max_results", 5),
                10  # Максимум 10 результатов
            )
            return {
                "query": tool_input.get("query", ""),
                "max_results": max_results
            }
        
        return tool_input
```

```python
class ToolRouter:

                "requires": ["query", "access_key"],

                "description": "Call external service",
                "requires": ["endpoint", "method"],

            }
        }

        )
```

### Паттерн 8: Parallel Tool Execution

Когда несколько инструментов независимы — выполняем параллельно.

```python
import asyncio
from concurrent.futures import ThreadPoolExecutor

class ParallelToolExecutor:
    """
    Выполняет несколько инструментов параллельно.
    
    Сокращает время выполнения при наличии нескольких независимых action.
    """
    
    def __init__(self, max_workers: int = 5):
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
    
    async def execute_parallel(self, 
                              tool_calls: list) -> Dict[str, Any]:
        """
        Выполнить несколько инструментов одновременно.
        
        tool_calls: список {"tool": "name", "input": {...}}
        """
        
        tasks = [
            self._execute_single(tool_call)
            for tool_call in tool_calls
        ]
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        return {
            call["tool"]: (
                results[i] 
                if not isinstance(results[i], Exception) 
                else {"error": str(results[i])}
            )
            for i, call in enumerate(tool_calls)
        }
    
    async def _execute_single(self, tool_call: Dict) -> Any:
        """Выполнить одиночный инструмент"""
        
        loop = asyncio.get_event_loop()
        
        # Выполняем в executor (не блокируем основной loop)
        return await loop.run_in_executor(
            self.executor,
            self._call_tool,
            tool_call["tool"],
            tool_call["input"]
        )
    
    def _call_tool(self, tool_name: str, tool_input: Dict) -> Any:
        """Выполнить инструмент (синхронно)"""
        
        # Реальная реализация зависит от инструмента
        if tool_name == "search":
            return search_api(tool_input["query"])
        elif tool_name == "get_weather":
            return weather_api(tool_input["location"])
        # ... и т.д.
```

## Часть 6: Метрики и оценка качества

### Метрики для агентов

```python
from dataclasses import dataclass
from typing import List

@dataclass
class AgentMetrics:
    """Метрики для оценки качества агента"""
    
    total_requests: int = 0
    successful_requests: int = 0
    failed_requests: int = 0
    total_tool_calls: int = 0
    average_tool_calls_per_request: float = 0.0
    average_tokens_per_request: int = 0
    average_cost_per_request: float = 0.0
    average_latency: float = 0.0  # в секундах
    tool_success_rate: float = 0.0
    hallucination_rate: float = 0.0  # % запросов с hallucinations

class AgentEvaluator:
    """
    Оценивает качество работы агента на основе метрик.
    """
    
    def __init__(self):
        self.metrics = AgentMetrics()
        self.request_log = []
    
    def evaluate_response(self, 
                         user_query: str,
                         agent_response: str,
                         tool_calls_made: int,
                         tokens_used: int,
                         cost: float,
                         latency: float,
                         is_correct: bool) -> dict:
        """
        Оценить один запрос.
        
        is_correct: верен ли ответ (определяется человеком или ground truth)
        """
        
        self.metrics.total_requests += 1
        self.metrics.total_tool_calls += tool_calls_made
        
        if is_correct:
            self.metrics.successful_requests += 1
        else:
            self.metrics.failed_requests += 1
        
        # Проверка на hallucinations
        has_hallucination = self._detect_hallucination(
            user_query, agent_response
        )
        
        # Записываем запрос
        request_record = {
            "query": user_query,
            "response": agent_response,
            "tool_calls": tool_calls_made,
            "tokens": tokens_used,
            "cost": cost,
            "latency": latency,
            "correct": is_correct,
            "hallucination": has_hallucination
        }
        self.request_log.append(request_record)
        
        # Пересчитываем средние значения
        self._update_averages()
        
        return {
            "success_rate": (
                self.metrics.successful_requests / 
                max(1, self.metrics.total_requests)
            ) * 100,
            "average_tool_calls": (
                self.metrics.total_tool_calls / 
                max(1, self.metrics.total_requests)
            ),
            "hallucination_detected": has_hallucination
        }
    
    def _detect_hallucination(self, query: str, response: str) -> bool:
        """
        Простая эвристика для обнаружения hallucinations.
        
        На практике используйте ML-модель или human evaluation.
        """
        
        # Если ответ содержит "I don't know" или "I cannot" — это OK
        if any(phrase in response.lower() for phrase in 
               ["i don't know", "i cannot", "i'm not sure"]):
            return False
        
        # Если ответ слишком специфичен, но в query нет деталей
        if len(response) > 1000 and len(query) < 100:
            return True  # возможно hallucination
        
        return False
    
    def _update_averages(self):
        """Обновить средние значения"""
        if not self.request_log:
            return
        
        self.metrics.average_tool_calls_per_request = (
            sum(r["tool_calls"] for r in self.request_log) / 
            len(self.request_log)
        )
        
        self.metrics.average_tokens_per_request = int(
            sum(r["tokens"] for r in self.request_log) / 
            len(self.request_log)
        )
        
        self.metrics.average_cost_per_request = (
            sum(r["cost"] for r in self.request_log) / 
            len(self.request_log)
        )
        
        self.metrics.average_latency = (
            sum(r["latency"] for r in self.request_log) / 
            len(self.request_log)
        )
        
        hallucinations = sum(
            1 for r in self.request_log 
            if r["hallucination"]
        )
        self.metrics.hallucination_rate = (
            hallucinations / len(self.request_log)
        ) * 100
    
    def get_report(self) -> str:
        """Получить отчёт о качестве агента"""
        
        report = f"""
=== Agent Performance Report ===

Requests:
  Total: {self.metrics.total_requests}
  Successful: {self.metrics.successful_requests}
  Failed: {self.metrics.failed_requests}
  Success Rate: {(self.metrics.successful_requests / max(1, self.metrics.total_requests) * 100):.1f}%

Tool Usage:
  Total tool calls: {self.metrics.total_tool_calls}
  Avg per request: {self.metrics.average_tool_calls_per_request:.2f}

Tokens & Cost:
  Avg tokens per request: {self.metrics.average_tokens_per_request}
  Avg cost per request: ${self.metrics.average_cost_per_request:.4f}

Latency:
  Average: {self.metrics.average_latency:.2f}s

Quality Issues:
  Hallucination rate: {self.metrics.hallucination_rate:.1f}%
"""
        return report
```

```python
def optimize_for_tokens(messages: list, max_tokens_for_history: int = 4000):
    """
    Сжимаем историю сообщений, если они занимают слишком много токенов.
    """
    
    # Примерная оценка: 1 токен ≈ 4 символа
    estimated_tokens = sum(len(m["content"]) // 4 for m in messages)
    
    if estimated_tokens > max_tokens_for_history:
        # Оставляем только последние сообщения
        messages = messages[-5:]  # Последние 5 сообщений
    
    return messages
```

```python
import json
from datetime import datetime

class ProductionAgentMonitor:
    """
    Мониторит работу агента в production.
    Отслеживает ошибки, latency, стоимость.
    """
    
    def __init__(self, alert_threshold_error_rate=0.1):
        self.error_count = 0
        self.total_count = 0
        self.alert_threshold = alert_threshold_error_rate
    
    def log_request(self, request_data: dict):
        """Логировать запрос в production"""
        
        self.total_count += 1
        
        if request_data.get("error"):
            self.error_count += 1
        
        # Пишем в файл или отправляем в мониторинг систему
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "request": request_data.get("query", "")[:100],  # первые 100 символов
            "success": not request_data.get("error"),
            "latency": request_data.get("latency"),
            "cost": request_data.get("cost"),
            "tool_calls": request_data.get("tool_calls", 0)
        }
        
        # Сохраняем
        with open("agent_production.log", "a") as f:
            f.write(json.dumps(log_entry) + "\n")
        
        # Проверяем пороги
        self._check_alerts()
    
    def _check_alerts(self):
        """Проверить, не превышены ли пороги"""
        
        current_error_rate = self.error_count / max(1, self.total_count)
        
        if current_error_rate > self.alert_threshold:
            self._send_alert(
                f"Agent error rate: {current_error_rate * 100:.1f}% "
                f"(threshold: {self.alert_threshold * 100:.1f}%)"
            )
    
    def _send_alert(self, message: str):
        """Отправить alert в систему мониторинга"""
        # Интеграция с DataDog, Prometheus, PagerDuty и т.д.
        print(f"ALERT: {message}")
```

## Приложение: Полный пример рабочего агента

```python
#!/usr/bin/env python3
"""
Полный пример рабочего агента с инструментами,
ошибка обработкой и логированием.
"""

from anthropic import Anthropic
import json
import time
from typing import Optional

class ProductionAgent:
    def __init__(self):
        self.client = Anthropic()
        self.model = "claude-3-5-sonnet-20241022"
        self.tools = self._define_tools()
        self.max_iterations = 10
        self.total_cost = 0.0
    
    def _define_tools(self):
        return [
            {
                "name": "calculator",
                "description": "Perform mathematical calculations",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "expression": {"type": "string"}
                    },
                    "required": ["expression"]
                }
            },
            {
                "name": "search",
                "description": "Search for information",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "query": {"type": "string"}
                    },
                    "required": ["query"]
                }
            }
        ]
    
    def _execute_tool(self, tool_name: str, tool_input: dict) -> str:
        """Выполнить инструмент"""
        
        if tool_name == "calculator":
            try:
                result = eval(tool_input["expression"])
                return f"Result: {result}"
            except Exception as e:
                return f"Error: {str(e)}"
        
        elif tool_name == "search":
            # Мок для демонстрации
            return f"Search results for '{tool_input['query']}':\n" \
                   "[Mock result 1]\n[Mock result 2]"
        
        return "Unknown tool"
    
    def run(self, user_message: str) -> str:
        """Выполнить агента"""
        
        messages = [
            {"role": "user", "content": user_message}
        ]
        
        system_prompt = """You are a helpful assistant with access to tools.
Use tools when needed to answer questions accurately."""
        
        for iteration in range(self.max_iterations):
            print(f"\n[Iteration {iteration + 1}]")
            
            # Вызываем модель
            start_time = time.time()
            response = self.client.messages.create(
                model=self.model,
                max_tokens=1024,
                system=system_prompt,
                tools=self.tools,
                messages=messages
            )
            latency = time.time() - start_time
            
            # Учитываем стоимость (примерно)
            self.total_cost += latency * 0.001
            
            # Проверяем, завершено ли
            if response.stop_reason == "end_turn":
                for block in response.content:
                    if hasattr(block, "text"):
                        return block.text
            
            # Обрабатываем tool use
            if response.stop_reason == "tool_use":
                messages.append({"role": "assistant", "content": response.content})
                
                tool_results = []
                for block in response.content:
                    if block.type == "tool_use":
                        tool_name = block.name
                        tool_input = block.input
                        
                        print(f"  Tool: {tool_name}")
                        print(f"  Input: {tool_input}")
                        
                        result = self._execute_tool(tool_name, tool_input)
                        print(f"  Result: {result[:100]}...")
                        
                        tool_results.append({
                            "type": "tool_result",
                            "tool_use_id": block.id,
                            "content": result
                        })
                
                messages.append({"role": "user", "content": tool_results})
        
        return "Max iterations reached"


if __name__ == "__main__":
    agent = ProductionAgent()
    
    # Примеры запросов
    queries = [
        "What is 25 * 4?",
        "Can you search for information about LLM agents?",
    ]
    
    for query in queries:
        print(f"\n{'='*60}")
        print(f"User: {query}")
        print(f"{'='*60}")
        
        result = agent.run(query)
        print(f"\nAgent: {result}")
        print(f"Total cost so far: ${agent.total_cost:.4f}")
```

Этот пример готов к использованию и демонстрирует все ключевые паттерны.
