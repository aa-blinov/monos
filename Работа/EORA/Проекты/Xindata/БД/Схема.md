## Примеры SQL

```
-- Выручка за день по маркетплейсу
SELECT
    COALESCE(SUM(CASE WHEN "streamType" = 0 THEN "amountValue" ELSE 0 END), 0) AS "revenue"
FROM v_ai_documents
WHERE
    LOWER("integrationtype") = LOWER('{marketplace}')
    AND "operationDate"::date = CURRENT_DATE - INTERVAL '1 day'
    AND "paymentPurpose" NOT IN (
        'Скидка постоянного покупателя (Продажа/Продажа)',
        'Скидка постоянного покупателя (Возврат/Возврат)'
    );
```

```
-- Выручка за день по wildberries
SELECT
    COALESCE(SUM("salesum"), 0) AS "revenue"
FROM v_ai_moneysellers_sale
WHERE
    LOWER("integrationtype") = 'wildberries'
    AND "saledate"::date = CURRENT_DATE - INTERVAL '1 day';
```

```
-- Топ товаров за день
SELECT
    provider,
    SUM(ordersum) AS total_sum
FROM
    (
        SELECT
            CASE
                WHEN "articleozonprovider" IS NOT NULL AND "articleozonprovider" <> '' THEN "articleozonprovider"
                WHEN "articleyandexmarketprovider" IS NOT NULL AND "articleyandexmarketprovider" <> '' THEN "articleyandexmarketprovider"
                WHEN "articlewbprovider" IS NOT NULL AND "articlewbprovider" <> '' THEN "articlewbprovider"
            END AS provider,
            "ordersum"
        FROM v_ai_moneysellers_order
        WHERE
            "orderdate"::date = CURRENT_DATE - INTERVAL '1 day'
            AND (
                ("articleozonprovider" IS NOT NULL AND "articleozonprovider" <> '') OR
                ("articleyandexmarketprovider" IS NOT NULL AND "articleyandexmarketprovider" <> '') OR
                ("articlewbprovider" IS NOT NULL AND "articlewbprovider" <> '')
            )
    ) AS combined_providers
WHERE
    provider IS NOT NULL
GROUP BY
    provider
ORDER BY
    total_sum DESC
LIMIT 10;
```

SELECT
    COALESCE(o.date, s.date, r.date) AS date,
    COALESCE(o.marketplace, s.marketplace, r.marketplace) AS marketplace,
    COALESCE(o.orders_count, 0) AS orders_count,
    COALESCE(o.orders_sum_rub, 0) AS orders_sum_rub,
    COALESCE(s.sales_count, 0) AS sales_count,
    COALESCE(s.sales_sum_rub, 0) AS sales_sum_rub,
    COALESCE(r.revenue, 0) - COALESCE(r.marketplace_costs, 0) AS margin,
    CASE
        WHEN COALESCE(r.revenue, 0) = 0 THEN 0
        ELSE (COALESCE(r.revenue, 0) - COALESCE(r.marketplace_costs, 0)) / COALESCE(r.revenue, 0) * 100
    END AS margin_percent
FROM
    (
        SELECT
            "orderdate"::DATE AS "date",
            LOWER("integrationtype") AS marketplace,
            COUNT(*) AS orders_count,
            SUM("ordersum") AS orders_sum_rub
        FROM v_ai_moneysellers_order
        WHERE
            "orderdate" >= DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow') - INTERVAL '4 weeks'
            AND "orderdate" < DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow')
        GROUP BY
            "orderdate"::DATE,
            LOWER("integrationtype")
    ) o
FULL OUTER JOIN
    (
        SELECT
            "operationDate"::DATE AS "date",
            LOWER("integrationtype") AS marketplace,
            COALESCE(SUM(CASE WHEN "streamType" = 0 THEN "quantityValue" ELSE 0 END), 0) AS "sales_count",
            COALESCE(SUM(CASE WHEN "streamType" = 0 THEN "amountValue" ELSE 0 END), 0) AS "sales_sum_rub"
        FROM v_ai_documents
        WHERE
            LOWER("integrationtype") IN ('ozon', 'wildberries', 'яндекс маркет')
            AND "operationDate" >= DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow') - INTERVAL '4 weeks'
            AND "operationDate" < DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow')
            AND "paymentPurpose" NOT IN (
                'Скидка постоянного покупателя (Продажа/Продажа)',
                'Скидка постоянного покупателя (Возврат/Возврат)'
            )
        GROUP BY
            "operationDate"::DATE,
            LOWER("integrationtype")
    ) s
ON o.date = s.date AND o.marketplace = s.marketplace
FULL OUTER JOIN
    (
        SELECT
            "operationDate"::DATE AS "date",
            LOWER("integrationtype") AS marketplace,
            COALESCE(SUM(CASE WHEN "streamType" = 0 THEN "amountValue" ELSE 0 END), 0) AS revenue,
            SUM(CASE WHEN "streamType" = 1 THEN "amountValue" ELSE 0 END) AS marketplace_costs
        FROM v_ai_documents
        WHERE
            LOWER("integrationtype") IN ('ozon', 'wildberries', 'яндекс маркет')
            AND "operationDate" >= DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow') - INTERVAL '4 weeks'
            AND "operationDate" < DATE_TRUNC('week', CURRENT_DATE AT TIME ZONE 'Europe/Moscow')
            AND "paymentPurpose" NOT IN (
                'Скидка постоянного покупателя (Продажа/Продажа)',
                'Скидка постоянного покупателя (Возврат/Возврат)',
                'Возмещение СПП (Продажа/Продажа)',
                'Возмещение СПП (Возврат/Возврат)'
            )
        GROUP BY
            "operationDate"::DATE,
            LOWER("integrationtype")
    ) r
ON COALESCE(o.date, s.date) = r.date AND COALESCE(o.marketplace, s.marketplace) = r.marketplace
WHERE
    COALESCE(o.marketplace, s.marketplace, r.marketplace) IS NOT NULL
ORDER BY
    date ASC,
    marketplace ASC;
